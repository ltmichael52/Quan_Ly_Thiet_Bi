USE [QLTHIETBI]
CREATE TABLE SINHVIEN (
	MASV INT PRIMARY KEY NOT NULL,
	TENSV NVARCHAR(255) NOT NULL,
	KHOA NVARCHAR (255) NOT NULL,
	NGANH NVARCHAR (255) NOT NULL,
	EMAIL NVARCHAR(255) NOT NULL,
	SDT NVARCHAR(255) NOT NULL,
	NGAYSINH DATETIME NOT NULL,
	DIACHI NVARCHAR(255)
);

CREATE TABLE NHANVIEN(
	MANV INT PRIMARY KEY IDENTITY(1,1),
	TENNV NVARCHAR(255) NOT NULL,
	SDT NVARCHAR(255) NOT NULL,
	PHAI NVARCHAR(3) NOT NULL,
	NGAYSINH DATETIME NOT NULL,
	EMAIL NVARCHAR(255) NOT NULL,
	DIACHI NVARCHAR(255)
);


CREATE TABLE COSO(
	MACS INT PRIMARY KEY IDENTITY(1,1),
	TENCS NVARCHAR(255) NOT NULL
);

CREATE TABLE PHONG(
	MAP NVARCHAR(30) PRIMARY KEY NOT NULL,
	MACS INT NOT NULL,
	TENPHONG NVARCHAR(255) NOT NULL,
	LOAIPHONG NVARCHAR(255) NOT NULL,
	FOREIGN KEY(MACS) REFERENCES COSO(MACS)
);

CREATE TABLE THIETBI(
	MATB NVARCHAR(255) PRIMARY KEY,
	TENTHIETBI NVARCHAR(255) NOT NULL,
	MAP NVARCHAR(30) NOT NULL,
	TRANGTHAI NVARCHAR(10) NOT NULL,
	FOREIGN KEY(MAP) REFERENCES PHONG(MAP)	
);

CREATE TABLE PHIEUMUON(
	MAPM INT PRIMARY KEY IDENTITY(1,1),
	NGAYLAP DATETIME NOT NULL,

);

CREATE TABLE CHITIETPHIEUMUON(
	MACTPM INT PRIMARY KEY IDENTITY(1,1),
	MAPM INT NOT NULL,
	MATB INT NOT NULL
);
